function adicionarCelular() { let nome = prompt("Nome do celular:"); let dpi = Number(prompt("DPI:")); let geral = Number(prompt("Geral:")); let redDot = Number(prompt("Red Dot:")); let mira2x = Number(prompt("Mira 2x:")); let mira4x = Number(prompt("Mira 4x:")); let awm = Number(prompt("AWM:")); let olhadinha = Number(prompt("Olhadinha:")); let botao = Number(prompt("Botão:")); dispositivos[nome] = { dpi, geral, redDot, mira2x, mira4x, awm, olhadinha, botao }; localStorage.setItem("dispositivos", JSON.stringify(dispositivos)); atualizarLista(); alert("Celular adicionado!"); }
