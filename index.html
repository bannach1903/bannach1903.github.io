// GERADOR DE SENSIBILIDADE FREE FIRE PRO // Autor: Aryon Script FF const dispositivos = { "Motorola G23": { dpi: 700, botaoDisparo: 58, sensibilidade: { geral: 95, redDot: 90, mira2x: 85, mira4x: 75, awm: 55, olhadinha: 70 } }, "Xiaomi Redmi Note 13 4G": { dpi: 720, botaoDisparo: 60, sensibilidade: { geral: 100, redDot: 95, mira2x: 90, mira4x: 80, awm: 60, olhadinha: 75 } }, "Samsung A14": { dpi: 680, botaoDisparo: 57, sensibilidade: { geral: 92, redDot: 88, mira2x: 82, mira4x: 72, awm: 52, olhadinha: 68 } } }; // CONFIG SISTEMA const configSistema = { velocidadeCursor: "rápida", escalaAnimacaoJanela: 0.5, escalaAnimacaoTransicao: 0.5, escalaDuracaoAnimador: 0.5 }; // AJUSTE POR ESTILO function aplicarEstilo(sens, estilo) { let mod = { ...sens }; switch (estilo) { case "rush": mod.geral += 5; mod.redDot += 5; break; case "capa": mod.redDot += 7; mod.mira2x += 5; break; case "sniper": mod.awm += 8; mod.mira4x += 5; break; } return mod; } // GERADOR function gerarSensibilidade(dispositivo, estilo = "padrao") { const dados = dispositivos[dispositivo]; if (!dados) { return { erro: "Dispositivo não encontrado!" }; } const sensFinal = aplicarEstilo(dados.sensibilidade, estilo); return { dispositivo, dpi: dados.dpi, botaoDisparo: dados.botaoDisparo, sensibilidade: sensFinal, sistema: configSistema }; } // TESTE const resultado = gerarSensibilidade("Xiaomi Redmi Note 13 4G", "capa"); console.log(JSON.stringify(resultado, null, 2));
